###### Functions

```javascript
  // Function-style
  function  functionName(arg1, arg2) {
    // code block
  }

  // Expression-style
  const functionName = function(arg1, arg2) {
    // code block
  }

  // Fat Arrow-style (ES6)
  const functionName = (arg1, arg2) => {
    // code block
  }

  // invoke a function with ()
  functionName(); //=> some result

```
